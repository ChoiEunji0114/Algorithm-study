//  2667.cpp

#include <stdio.h>
#include <iostream>
#include <queue>
#include <algorithm>

using namespace std;

int n;

int map[25][25];
int ans[25][25];

const int dx[] = {0,0,-1,1};
const int dy[] = {-1,1,0,0};

queue <pair<int,int>> que;
int cntArr[25];

void bfs(){
    int k = 0;
    int cur_y, cur_x, ny, nx, cnt;
    
    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
            
            if(map[i][j]==1 && ans[i][j]==0){
                k++;
                cur_y = i;
                cur_x = j;
                que.push(make_pair(cur_y, cur_x));
                ans[cur_y][cur_x] = k;
                cnt = 1;
                
                while(!que.empty()){
                    cur_y = que.front().first;
                    cur_x = que.front().second;
                    que.pop();
                    
                    for(int i=0; i<4; i++){
                        ny = cur_y + dy[i];
                        nx = cur_x + dx[i];
                        
                        if(ny<0 || nx<0 || ny>=n || nx>=n)
                            continue;
                        
                        if(map[ny][nx]==0)
                            continue;
                        
                        if(map[ny][nx]==1 && ans[ny][nx]==0){
                            ans[ny][nx] = k;
                            que.push(make_pair(ny, nx));
                            cnt++;
                        }
                    }
                }
            }
        }
    }
    cout << "k : " << k << "\n\n";
}


int main(void){
    
    cin>>n;
    
    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
            scanf("%1d", &map[i][j]);
        }
    }
    
    bfs();
    cout << "\n";
    
    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
            cout << ans[i][j] << " ";
        }
        cout << "\n";
    }
    
    
    return 0;
}
